    @NonNull
    private final String name;
    @NonNull
    private final String fullName;

    FormatEnum(@NonNull final String name, @NonNull final String fullName) {
        this.name = name;
        this.fullName = fullName;
    }

    /**
     * Get the name of the ENUM.
     *
     * @return the name of the ENUM
     */
    @NonNull
    public String getName() {
        return this.name;
    }


    /**
     * Get the full name of the ENUM.
     *
     * @return the full name of the ENUM
     */
    @NonNull
    public String getFullName() {
        return this.fullName;
    }

    /**
     * Get FormatEnum corresponding to given string.
     *
     * @param param
     * @return corresponding FormatEnum
     */
    @Nullable
    public static final FormatEnum getFormatEnum(@Nullable final String param) {
        FormatEnum result = null;
        if (param != null) {
            final String trimmedParam = param.trim().toLowerCase();
            if (trimmedParam.length() > 0) {
                for (final FormatEnum codecEnum : FormatEnum.values()) {
                    if (trimmedParam.equals(codecEnum.getName())) {
                        result = codecEnum;
                        // found, just break
                        break;
                    }
                }
            }
        }
        return result;
    }
